/* Note Sizes */
.note-main { width: 100px; height: 100px; font-size: 18px; }

.note-grace { width: 50px; height: 50px; font-size: 11px; }

/* Borders */
.note { border: 4px solid; border-radius: 50%; }

/* Polka Dots */
.dots-pattern {
    background: radial-gradient(circle, color 40%, transparent 40%),
                radial-gradient(circle, color 40%, transparent 40%),
                #000000;
    background-size: 20px 20px;
    background-position: 0 0, 10px 10px;
}

/* Stripes */
.stripes-pattern {
    background-image: repeating-linear-gradient(
        90deg,
        transparent,
        transparent 3px,
        color 3px,
        color 6px
    );
}

/* Glows */
.static-glow { box-shadow: 0 0 20px rgba(color, 0.6); }
.neon-glow { box-shadow: 0 0 30px color, 0 0 60px rgba(color, 0.5); }

/* Pulse Animation */
@keyframes pulse {
    0%, 100% {
        transform: scale(1);
        box-shadow: 0 0 10px rgba(color, 0.5);
    }
    50% {
        transform: scale(1.1);
        box-shadow: 0 0 30px rgba(color, 0.8);
    }
